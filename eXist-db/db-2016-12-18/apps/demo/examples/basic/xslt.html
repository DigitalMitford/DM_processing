Calling XSLT transformation

Simple Transformation

xquery version "3.0";

let $input := doc("@@path/examples/basic/Items.xml")
let $xsl := doc("@@path/examples/basic/ConvertItems.xsl")
return
    transform:transform($input, $xsl, ())
